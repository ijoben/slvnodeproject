document.getElementById('copyRefLinkBtn').addEventListener('click', function() { const linkText = document.getElementById('infoRefLink').textContent; if (!linkText || linkText === '-') { showToast('❌ No referral link available.'); return; } if (navigator.clipboard) { navigator.clipboard.writeText(linkText).then(() => { this.classList.add('copied'); this.innerHTML = ' Copied!'; showToast('📋 Referral link copied!'); setTimeout(() => { this.classList.remove('copied'); this.innerHTML = ' Copy'; }, 2500); }).catch(() => { fallbackCopy(linkText); }); } else { fallbackCopy(linkText); } });
